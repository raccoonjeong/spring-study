function Counter() {
    // 내부에 state 영역이 있고 이 영역에 값을 저장, 0을 읽고 정수구나 정수를 저장할 공간을
    // 만들고 이 공간에 접근해서 값을 수정할 수 있는 함수를 만들어서 내부에 보관하고
    // 값(주소)을 던진다
        const [number, setNumber] = React.useState(0);
        // 값을 읽어오는 건 number인데 값을 바꾸는건 함수로 하는지?
        const increase = () => {
          setNumber(number + 1);
        };
        const decrease = () => {
          setNumber(number - 1);
        };
        // ui를 출력한다.
        return (
          <div>
            <h1>{number}</h1>
            <button type="button" onClick={increase}>
              +1
            </button>
            <button type="button" onClick={decrease}>
              -1
            </button>
          </div>
        );
      }
